simulare_variabila_aleatoare_finita <- function(valori, probabilitati) 
{
  u <- runif(1)
  acumulata <- 0
  for (i in 1:length(probabilitati)) 
  {
    acumulata <- acumulata + probabilitati[i]
    if (u < acumulata) 
    {
      return(valori[i])
    }
  }
}
valori <- c("x1", "x2", "x3")
probabilitati <- c(0.2, 0.3, 0.5)

simulare_variabila_aleatoare_finita(valori, probabilitati)
